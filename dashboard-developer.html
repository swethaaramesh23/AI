Quick Integration

Initialize the STACKLY AI client in your application.

import stackly

client = stackly.Client(
    api_key="your-api-key"
)

response = client.chat.create(
    model="stackly-v4",
    messages=[
        {"role": "user", "content": "Analyze this dataset"}
    ]
)

print(response.choices[0].message)
